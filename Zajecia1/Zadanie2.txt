  def recOdPoczatku(ints: List[String]): String = ints match { 
    case Nil => ""
    case x :: tail => x + ", "+ recOdPoczatku(tail)
  }

  def recOdKonca(ints: List[String]): String = ints match { 
    case Nil => ""
    case x :: head => x + ", "+ recOdKonca(head)
  }
  
  
println(recOdPoczatku(dniTygodnia))
